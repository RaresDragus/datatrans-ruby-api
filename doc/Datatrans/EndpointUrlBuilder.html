def initialize(args)
  @action = args[:action]
  @env = args[:env]
  @env_subdomain = @env == :live ? '' : '.sandbox'
  @id = args[:id]
  @service = args[:service]
  @version = args[:version]
  raise ArgumentError, 'Invalid id specified' if @id.blank? && action_requires_id?
end
